⚙️ CI/CD Pipeline Automation

Docker Build → Docker Hub → Kubernetes Deployment → Prometheus & Grafana
GitHub Jenkins Pipeline Docker Image Docker Hub Kubernetes (k8s) Prometheus + Grafana

Jenkinsfile

pipeline {
  agent any
  stages {
    stage('Checkout') { steps { git 'https://github.com/durgadevi/cicd-node-app' } }
    stage('Build Docker') { steps { sh 'docker build -t durgadevi/node-app:latest .' } }
    stage('Push to Docker Hub') { steps { sh 'docker push durgadevi/node-app:latest' } }
    stage('Deploy to K8s') { steps { sh 'kubectl apply -f k8s/deployment.yaml' } }
    stage('Monitor') { steps { sh 'kubectl port-forward svc/prometheus 9090' } }
   }
}
Multibranch pipeline configured

Dockerfile

FROM node:18-alpine
WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
EXPOSE 3000
CMD ["node", "server.js"]
Multi-stage ready • Slim image ~150MB

K8s Manifests

apiVersion: apps/v1
kind: Deployment
metadata: { name: node-app }
spec:
  replicas: 2
  template:
    spec:
      containers:
      - name: app
        image: durgadevi/node-app:latest
        ports: [containerPort: 3000]
---
apiVersion: v1
kind: Service (LoadBalancer)
